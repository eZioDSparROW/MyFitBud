// Load articles function function loadArticles() { fetch('data/articles.json') .then(response => response.json()) .then(articles => { const container = document.getElementById('articles-container'); articles.forEach(article => { container.innerHTML += `
${article.category}

${article.title}

${article.excerpt}

Read More
`; }); }); } // Call when page loads document.addEventListener('DOMContentLoaded', loadArticles);
